### 蛇形矩阵

```javascript
//蛇形矩阵
function getSnakeMatrix(num) {
    let dp = [];
    let n = 1;
    for(let i = 0; i < num; i++) {
        for(let j = i; j >= 0; j--) {
            if(!dp[j]) dp[j] = [];
            dp[j].push(n++)
        }
    }
    for(let i = 0; i < num; i++) {
        console.log(dp[i].join(' '))
    }
}
 
getSnakeMatrix(5)
//输出
1 3 6 10 15
2 5 9 14   
4 8 13     
7 12       
11
```
