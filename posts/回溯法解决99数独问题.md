const sudokuGrid = [
    [5, 3, 0, 0, 7, 0, 0, 0, 0],
    [6, 0, 0, 1, 9, 5, 0, 0, 0],
    [0, 9, 8, 0, 0, 0, 0, 6, 0],
    [8, 0, 0, 0, 6, 0, 0, 0, 3],
    [4, 0, 0, 8, 0, 3, 0, 0, 1],
    [7, 0, 0, 0, 2, 0, 0, 0, 6],
    [0, 6, 0, 0, 0, 0, 2, 8, 0],
    [0, 0, 0, 4, 1, 9, 0, 0, 5],
    [0, 0, 0, 0, 8, 0, 0, 7, 9],
]
 
function suduSolver(sudu) {
    if (solveSudoku(sudu)) {
        return sudu
    }
    return 'UNSOLVED'
}
 
const UNASSIGNED = 0
 
function solveSudoku(sudu) {
    let rowIndex = null, colIndex = null, checkBlankSpace = false
    //如果有空白位置我们要从两个循环中跳出
    for (let row = 0; row < 9; row++) {
        for (let col = 0; col < 9; col++) {
            if (sudu[row][col] === UNASSIGNED) {
                rowIndex = row
                colIndex = col
                checkBlankSpace = true
                break
            }
        }
 
        if (checkBlankSpace) {
            break
        }
    }
 
    if (!checkBlankSpace) {
        return true
    }
 
    for (let num = 1; num <= 9; num++) {
        //先给未分配值的元素赋值然后依次往后递归调用如果有不符合的就回溯
        //重新赋值为UNASSIGNED 未分配的状态然后进行下一轮循环
        if (isSafe(rowIndex, colIndex, sudu, num)) {
            sudu[rowIndex][colIndex] = num
            if (solveSudoku(sudu)) {
                return true
            }
            sudu[rowIndex][colIndex] = UNASSIGNED
 
        }
    }
    return false
 
}
//检查每一行是否符合规则
function isSafe(row, col, sudu, num) {
    return !usedInRow(row, sudu, num) &&
        !usedInCol(col, sudu, num) &&
        !usedInBox(row - row % 3, col - col % 3, sudu, num)
 
}
//检查每一列是否符合规则
function usedInRow(row, sudu, num) {
    for (let i = 0; i < 9; i++) {
        if (sudu[row][i] === num) {
            return true
        }
    }
    return false
}
//检查每一宫是否符合规则
function usedInCol(col, sudu, num) {
    for (let i = 0; i < 9; i++) {
        if (sudu[i][col] === num) {
            return true
        }
    }
    return false
}
 
function usedInBox(row, col, sudu, num) {
    for (let i = 0; i < 3; i++) {
        for (let j = 0; j < 3; j++) {
            if (sudu[row + i][col + j] === num) {
                return true
            }
        }
    }
    return false
 
}
console.log(suduSolver(sudokuGrid));
 
// 9x9数独规则
//  1.数字 1-9 在每一行只能出现一次。
// 	2.数字 1-9 在每一列只能出现一次。
// 	3.数字 1-9 在每一个以粗实线分隔的 3x3 宫内只能出现一次。
