//背包问题
function knapStack(capacity, weights, values) {
    const valueLen = values.length
    let load = 0
    let val = 0
    for (let i = 0; i < valueLen && load < capacity; i++) {
        if (weights[i] <= capacity - load) {
            val += values[i]
            load += weights[i]
        } else {
            const r = (capacity - load) / weights[i]
            val += r * values[i]
            load += weights[i]
        }
 
 
    }
    return val
}
